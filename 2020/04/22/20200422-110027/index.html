逆波兰计算器的实现

后缀表达式求值

在本程序中 后缀表达式使用ArrayList来进行存储

思路：

  1. 遍历ArrayList元素，遇到数字压栈，
  2. 遇到操作符，从栈中弹出两个操作数（第一个弹出的在操作符后面，第二个弹出的在操作符前面），进行计算，将结果重新入栈
  3. 循环2过程，直到元素遍历完成，最后栈中只剩下一个操作数
  4. 栈中的数字即为最后的表达式结果

字符串转整数：Integer.parseInt()

整数转字符串：num+"",进行数据类型的隐式转换

缀表达式也叫逆波兰表达式，其求值过程可以用到栈来辅助存储。假定待求值的后缀表达式为：6 5 2 3 + 8 * + 3 + *，则其求值过程如下：

1）遍历表达式，遇到的数字首先放入栈中，此时栈如下所示 ：

2）接着读到“+”，则弹出3和2，执行3+2，计算结果等于5，并将5压入到栈中 ：

3）读到8，将其直接放入栈中：

4）读到*，弹出8和5，执行8*5，并将结果40压入栈中。而后过程类似，读到“+”，将40和5弹出，将40+5的结果45压入栈…以此类推。最后求的值288。

代码实现

package com.homewyk.stack;

import java.util.ArrayList;
import java.util.Iterator;
import java.util.List;
import java.util.Stack;

public class suffixExpression {
    public static void main(String[] args) {
        System.out.println("逆波兰计算器的简单实现");
        String  suffixExpressionString = "1 2 3 + 4 x + 5 -";
        SuffixCalculator calc = new SuffixCalculator(suffixExpressionString);
        System.out.println(calc.calc());
    }
}

/**
 * 1. 直接计算后缀表达式
 * 2. 字符串形式："1 2 3 + 4 x + 5 -" 中间使用空格进行分割
 * @author asdw1
 *
 */
class SuffixCalculator{
    private String suffixExpressionString;

    private List<String> suffixListExpression = new ArrayList<String>(); 

    Stack<String> stack = new Stack<String>();

    public SuffixCalculator(String suffixExpressionString) {
        this.suffixExpressionString = suffixExpressionString;
    }

    private void StringConvertList() {
        String[] expresses = suffixExpressionString.split(" ");
        for (String string : expresses) {
            suffixListExpression.add(string);
        }
    }

    public int calc() {
        StringConvertList();

        for(Iterator<String> it = suffixListExpression.iterator(); it.hasNext();) {
            String ch = it.next();
            ///  如果是数字，入栈
            if((ch.charAt(0) >= 48)&&(ch.charAt(0) <= 57)) {
                stack.push(ch);
            }else { // 如果是操作符，则抛出2个数字，并且进行计算将结果入栈
                int num2 = Integer.parseInt(stack.pop());;
                int num1 = Integer.parseInt(stack.pop());
                int result = 0;
                switch (ch.charAt(0)) {
                case '+':
                    result = num1 + num2;
                    break;
                case '-':
                    result = num1 - num2;
                    break;
                case 'x':
                    result = num1 * num2;
                    break;
                case '/':
                    result = num1 / num2;
                    break;
                default:
                    throw new RuntimeException("无效操作符");
                }

                stack.push(result + "");
            }
        }
        return Integer.parseInt(stack.pop());
    }
}

中缀表达式转后缀表达式

中缀表达式$ a + bc + (d e + f) g $，其转换成后缀表达式则为$ a b c + d e f + g + $。

转换过程需要用到栈，具体过程如下：

  1. 初始化两个栈：运算符栈$S1$ 和 存储中间结果的栈$S2$

  2. 扫描顺序，从左到右

  3. 如果遇到多位字符串，直接入栈$S2​$

    1. 遇到单字符，进行判断

      1. 单字符为数字，直接入栈$S2$

      2. 单字符为操作符：

        1. 如果$S1$为空,直接入栈$S1$

        2. $S1$的栈顶为$($，直接入栈$S1$

        3. 如果$S1$非空且$S1$栈顶元素非$($，此时判断此时的运算符和栈顶运算符的优先级：

          1. 此时操作符的优先级 > 栈顶的优先级 =====》 直接入栈
          2. 否则此时操作符的优先级 <= 栈顶的优先级 =====》 进行遍历操作
            1. 判断操作栈$S1$不为空且此时操作符的优先级 <= 栈顶的优先级 并且栈顶元素不为$($:
            2. $S1$栈顶元素出栈，然后入栈到$S2$中
            3. 返回与此缩进相同的$1$

        4. 遇到$)$

          1. 将$S1$中的运算符出栈到S2中，

          2. while的结束条件为栈顶元素为$)$

          3. 循环结束后且需要将$)$出栈

            在进行编程的时候，我将这一步提到了最前面！！

  4. 重复步骤$3$，直到表达式的最后

  5. 将$S1$的剩余部分依次入栈到$S2$

  6. 依次弹出$S2$中的元素并且进行输出，结果的逆序即为中缀表达式所对应的后缀表达式。

具体实现

第一部分：

将中缀表达式拆分出来，存储到ArrayList中

还是要注意多位数的处理为题

第二部分: 中缀表达式到后缀表达式的逻辑转换

代码实现

ppackage com.homewyk.stack;

import java.util.ArrayList;
import java.util.Iterator;
import java.util.List;
import java.util.Stack;

public class suffixExpression {
    public static void main(String[] args) {

        SuffixCalculator calc = new SuffixCalculator(suffixExpressionString);
        System.out.println(calc.calc());

        String infixExpressionString = "(1+2)/3 + (5*2)";
        InfixConvertSufix ics = new InfixConvertSufix(infixExpressionString);
        ics.StringConvertList();
        ics.showInfixList();

        System.out.println("中缀表达式转换为后缀表达式为： " + ics.convert());
        SuffixCalculator calc1 = new SuffixCalculator(ics.convert());
        System.out.println("计算结果为：" + calc1.calc());



    }

/**
 * 1. 实现中缀表达式转为后缀表达式
 *     1.1 将中缀表达式存入ArrayList中
 *         1.1.1 需要注意的多位数的处理问题
 *  1.2 实现中缀表达式转为后缀表达式
 * @author asdw1
 *
 */
class InfixConvertSufix {
    //接收表达式
    private String infixExpressionString;
    //中缀表达式存储到list中，这里已经实现了多位数的处理
    private List<String> infixList = new ArrayList<String>(); 
    //表达式指针
    private int index = 0;
    //控制循环
    private boolean loop = true; 
    //中缀表达式多位数的组合
    StringBuilder sb = new StringBuilder(); 
    //中转后的两个stack
    Stack<String> operStack = new Stack<String>();
    Stack<String> resultStack = new Stack<String>();


    public InfixConvertSufix(String infixExpressionString){

        this.infixExpressionString = infixExpressionString;

    }
    /**
     * 1. 实现中缀表达式进行元素拆分并且存入arraylist中
     * 2. 已经处理了多位数：
     *     2.1 如果index到了最后，直接添加
     *     2.2 判断index+1是否是数字，用来确定是否添加
     *     2.3 注意在添加到ArrayList之后要删除下
     */
    public void StringConvertList() {
        while(loop) {
            //接收字符
            char ch = infixExpressionString.substring(index, index+1).charAt(0);
            // 字符如果是操作符 直接添加
            if((ch == '+')||(ch == '-')||(ch == '*')||
                (ch == '/')||(ch == 'x')||(ch == '(') || (ch == ')')) {
                infixList.add(ch+"");
            }else if((ch >= 48)&&(ch <= 57)){ //如果字符是数字，如果索引处于字符串最后，直接添加； 否则还要判断后一位的字符类型
                sb.append(ch);
                if(index == infixExpressionString.length() -1) {
                    infixList.add(sb.toString());
                    sb.delete(0, sb.length());
                }else if(infixExpressionString.substring(index+1, index+2).charAt(0) < 48 || 
                         infixExpressionString.substring(index+1, index+2).charAt(0) > 57) {
                    infixList.add(sb.toString());
                    sb.delete(0, sb.length());
                }
            }
            index ++;
            if(index >= infixExpressionString.length()) {
                loop = false;
            }
        }
    }
    /**
     * 1. 显示分割好的中缀表达式
     */
    public void showInfixList() {
        for (String string : infixList) {
            System.out.print(string);
        }
        System.out.println();
    }
    /**
     * 1.中转后的详细逻辑
     * @return
     */
    public String convert() {
        for (String item : infixList) {
            //如果item的长度 > 1,item一定是数字
            //这里是用于多位数的处理
            if(item.length() > 1) {
                resultStack.push(item);
            }else if(item.length() == 1) {
                char ch = item.charAt(0);
                //如果是数字 直接入栈
                if(!isOper(ch)) {
                    resultStack.push(ch+"");
                    //如果是运算符
                }else if(isOper(ch)){ 
                    // 如果遇到)
                    if(!operStack.isEmpty()&& (ch == ')')) {
                        while(operStack.peek().charAt(0) != '(') {
                            resultStack.push(operStack.pop());
                        }
                        //pop (
                        operStack.pop();
                        //如果操作栈为空
                    }else if (operStack.isEmpty()) {
                        operStack.push(ch+"");
                        //S1的栈顶为( ，直接入栈
                    }else if (ch == '(') {
                        operStack.push(ch+"");
                    }else if (operStack.peek().charAt(0) == '(') { 
                        operStack.push(ch+"");
                        // 如果S1不为空且S1的栈顶元素不为(， 进行操作符优先级的比较
                    }else if (!operStack.isEmpty()&& (operStack.peek().charAt(0) != '(')) { 
                        //如果操作符的优先级 > 栈顶的优先级 直接入栈
                        if(priority(ch) > priority(operStack.peek().charAt(0))) {
                            operStack.push(ch+"");
                            //如果操作符的优先级 <= 栈顶的优先级 且栈顶操作符不为(，操作符出栈
                            //当前的操作符入栈
                        }else { 
                            //！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！！
                            //出现错误的地方
                            //修改了！ 必须要加这个判断！！(!operStack.isEmpty())
                            while((!operStack.isEmpty())&&(operStack.peek().charAt(0) != '(')&&(priority(ch) <= priority(operStack.peek().charAt(0)))) {
                                resultStack.push(operStack.pop());
                            }
                            operStack.push(ch+"");
                        }
                    }
                }
            }
        }
        //循环完毕,将操作数依次出栈到结果栈中
        while(!operStack.isEmpty()) {
            resultStack.push(operStack.pop());
        }
        StringBuilder sb = new StringBuilder();
        while(!resultStack.isEmpty()) {
            //后缀表达式约定是靠空格区分的
            sb.append(' ');
            sb.append(resultStack.pop());
        }
        sb.reverse();
        return sb.toString();

    }
    /**
     * 1. 判断是否是操作符
     * @param ch
     * @return
     */
    public boolean isOper(char ch) {
        if ((ch == '+')||(ch == '-')||(ch == '/')||(ch =='*')||(ch == '(')||(ch == ')')) {
            return true;
        }
        return false;
    }

    public int priority(char ch) {
        if((ch == '+') || (ch == '-')) {
            return 1;
        }else if ((ch == '*') || (ch == '/')) {
            return 2;
        }
        return -1;
    }
}



/**
 * 1. 直接计算后缀表达式
 * 2. 字符串形式："1 2 3 + 4 x + 5 -" 中间使用空格进行分割
 * @author asdw1
 *
 */
class SuffixCalculator{
    private String suffixExpressionString;

    private List<String> suffixListExpression = new ArrayList<String>(); 

    Stack<String> stack = new Stack<String>();

    public SuffixCalculator(String suffixExpressionString) {
        this.suffixExpressionString = suffixExpressionString;
    }
    /**
     *  1.后缀表达式存放在ArrayList中
     */
    private void StringConvertList() {
        String[] expresses = suffixExpressionString.split(" ");
        for (String string : expresses) {
            suffixListExpression.add(string);
        }
    }
    /**
     * 1. 进行计算
     * @return
     */
    public int calc() {
        StringConvertList();

        for(Iterator<String> it = suffixListExpression.iterator(); it.hasNext();) {
            String ch = it.next();
            ///  如果是数字，入栈
            if((ch.charAt(0) >= 48)&&(ch.charAt(0) <= 57)) {
                stack.push(ch);
            }else { // 如果是操作符，则抛出2个数字，并且进行计算将结果入栈
                int num2 = Integer.parseInt(stack.pop());;
                int num1 = Integer.parseInt(stack.pop());
                int result = 0;
                switch (ch.charAt(0)) {
                case '+':
                    result = num1 + num2;
                    break;
                case '-':
                    result = num1 - num2;
                    break;
                case '*':
                    result = num1 * num2;
                    break;
                case '/':
                    result = num1 / num2;
                    break;
                default:
                    throw new RuntimeException("无效操作符");
                }

                stack.push(result + "");
            }
        }
        return Integer.parseInt(stack.pop());
    }
}
