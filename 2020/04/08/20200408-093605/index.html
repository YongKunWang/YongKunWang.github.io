二维数组和稀疏数组的相互转化

实际情况

实际需求：

五子棋程序的存盘和复盘操作

棋盘大小：$11\times 11$

棋盘黑子：用$1$表示

棋盘白子：用$2$表示

使用二维数组直接来保存棋盘数据：

问题：该二维数组的很多只都是默认为0的，因此记录了很多的无意义数据

可以使用稀疏数组进行解决

稀疏数组

当一个数组中的元素大部分为0或者相同值很多的时候，可以使用稀疏数组来保存该数组

二维数组 -> 稀疏数组

row col val
二维数组的行数 二维数组的列数 有几个不相同的值
第1个不同值的row 第1个不同值的col 第1个不同值的val
… … …
第n个不同值的row 第n个不同值的col 第n个不同值的val
  • 记录两大类信息：
    • 二维数组一共有几行几列，有多少个不同的值
    • 具体不同的值的行列值
  • 缩小了程序的规模

应用实例

  1. 使用稀疏数组，来保存棋盘信息
  2. 把稀疏数组存盘，并且可以恢复为原来的二维数组
  3. 整体思路分析：

二维数组 转 稀疏数组的思路

  1. 遍历 原始的二维数组，得到有效数据的个数 sum
  2. 根据sum 就可以创建 稀疏数组 sparseArr int[sum + 1][3]
  3. 将二维数组的有效数据数据存入到 稀疏数组

稀疏数组转原始的二维数组的思路

  1. 先读取稀疏数组的第一行，根据第一行的数据，创建原始的二维数组，比如上面的 chessArr2 = int [11][11]
  2. 在读取稀疏数组后几行的数据，并赋给 原始的二维数组 即可.

注意点

  1. 稀疏数组的第一行代表的含义
  2. 对于二维数组来说，稀疏数组只有3列
  3. 在编程是特别要注意对于第一行的处理
  4. 在使用eclipse做编码的时候，最好也不要出现警告！

实现代码

package com.homewyk.sparsearray;

import org.junit.Test;

/***
 * 1. 棋盘的保存和复盘
 * 二维数组 -> 稀疏数组 -> 二维数组
 * @author asdw1
 *
 */

public class SparseArray {

    @Test
    public void testSparse() {
        /* 
         * 1. 新建棋盘11*11，并设置2个黑白棋子 
         */
        int[][] chessArray1 = new int[11][11];
        chessArray1[1][2] = 1;
        chessArray1[2][3] = 2;

        System.out.println("输出原始棋盘布局");
        for (int[] rows : chessArray1) {
            for (int item : rows) {
                System.out.printf("%d\t",item);
            }
            System.out.println();
        }

        System.out.println();


        /*
         * 2. 转换为稀疏数组
         */
        //需要先得到稀疏数组的大小
        int sum = 0;
        for (int[] rows : chessArray1) {
            for (int item : rows) {
                if(item != 0)
                    sum ++;
            }
        }
        // 创建稀疏数组
        int[][] sparseArr = new int[sum+1][3];

        //稀疏数组第一行赋值
        sparseArr[0][0] = chessArray1.length;
        sparseArr[0][1] = chessArray1[0].length;
        sparseArr[0][2] = sum;

        //稀疏数组后续元素的赋值操作
        int count = 0;
        for(int i = 0; i < chessArray1.length; i++) {
            for (int j = 0; j < chessArray1[0].length; j++) {
                if(chessArray1[i][j]!= 0) {
                    count ++;
                    sparseArr[count][0] = i;
                    sparseArr[count][1] = j;
                    sparseArr[count][2] = chessArray1[i][j];
                }
            }
        }

        System.out.println("二维数组转换成的稀疏数组");
        for (int[] rows : sparseArr) {
            for (int item : rows) {
                System.out.printf("%d\t",item);
            }
            System.out.println();
        }

        System.out.println();
        /*
         * 3. 稀疏数组转换为棋盘
         */
        int[][] chessArray2 = new int[sparseArr[0][0]][sparseArr[0][1]];
        //注意要从1开始
        for (int i = 1; i < sparseArr.length; i++) {
            chessArray2[sparseArr[i][0]][sparseArr[i][1]] = sparseArr[i][2];
        }
        System.out.println("稀疏数组转换成的二维数组");
        for (int[] rows : chessArray2) {
            for (int item : rows) {
                System.out.printf("%d\t",item);
            }
            System.out.println();
        }
        System.out.println();

    }
}

保存和读取棋盘数据

  • 存盘操作使用bufferedWriter
  • 读取操作使用bufferedReader
  • 字符串转换为整形Integer.parseInt(String)
package com.homewyk.sparsearray;

import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.File;
import java.io.FileReader;
import java.io.FileWriter;

import org.junit.Test;

/***
 * 1. 棋盘的保存和复盘
 * 二维数组 -> 稀疏数组 -> 二维数组
 * 加入了从文件读取和保存到文件的操作
 * @author asdw1
 *
 */

public class SaveAndLoad {

    @Test
    public void testSparse() throws Exception {
        /* 
         * 1. 新建棋盘11*11，并设置2个黑白棋子 
         */
        int[][] chessArray1 = new int[11][11];
        chessArray1[1][2] = 1;
        chessArray1[2][3] = 2;

        System.out.println("输出原始棋盘布局");
        for (int[] rows : chessArray1) {
            for (int item : rows) {
                System.out.printf("%d\t",item);
            }
            System.out.println();
        }

        System.out.println();


        /*
         * 2. 转换为稀疏数组
         */
        //需要先得到稀疏数组的大小
        int sum = 0;
        for (int[] rows : chessArray1) {
            for (int item : rows) {
                if(item != 0)
                    sum ++;
            }
        }
        // 创建稀疏数组
        int[][] sparseArr = new int[sum+1][3];

        //稀疏数组第一行赋值
        sparseArr[0][0] = chessArray1.length;
        sparseArr[0][1] = chessArray1[0].length;
        sparseArr[0][2] = sum;

        //稀疏数组后续元素的赋值操作
        int count = 0;
        for(int i = 0; i < chessArray1.length; i++) {
            for (int j = 0; j < chessArray1[0].length; j++) {
                if(chessArray1[i][j]!= 0) {
                    count ++;
                    sparseArr[count][0] = i;
                    sparseArr[count][1] = j;
                    sparseArr[count][2] = chessArray1[i][j];
                }
            }
        }

        System.out.println("二维数组转换成的稀疏数组");
        for (int[] rows : sparseArr) {
            for (int item : rows) {
                System.out.printf("%d\t",item);
            }
            System.out.println();
        }

        System.out.println();

        /*
         * 存盘操作
         */
        File file = new File("map.dat");
        if(!file.exists()) {
            file.createNewFile();
        }
        BufferedWriter bw = new BufferedWriter(new FileWriter(file));
        for (int[] rows : sparseArr) {
            bw.write(rows[0]+"\t" + rows[1]+"\t"+rows[2]);
            bw.newLine();
            bw.flush();
        }
        bw.close();
        /*
         * 3. 稀疏数组转换为棋盘
         * 从文件中得到稀疏数组
         */
        BufferedReader br = new BufferedReader(new FileReader(file));
        String str = null;
        int cnt = 0;
        int[][] sparseArr2 = null;
        while((str = br.readLine()) != null) {
            String[] arr = str.split("\t");
            if(cnt == 0) {
                //要注意这一步！！！arr[2]得到的是不同数据的个数，未算第0行
                sparseArr2 = new int[Integer.parseInt(arr[2])+1][3];
                System.out.println(sparseArr2.length+sparseArr2[0].length);
            }
            sparseArr2[cnt][0] = Integer.parseInt(arr[0]);
            sparseArr2[cnt][1] = Integer.parseInt(arr[1]);
            sparseArr2[cnt][2] = Integer.parseInt(arr[2]);
            cnt ++;
        }
        br.close();
        System.out.println("从文件中解析的稀疏数组");
        for (int[] rows : sparseArr2) {
            for (int item : rows) {
                System.out.printf("%d\t",item);
            }
            System.out.println();
        }

        int[][] chessArray2 = new int[sparseArr2[0][0]][sparseArr2[0][1]];
        //注意要从1开始
        for (int i = 1; i < sparseArr2.length; i++) {
            chessArray2[sparseArr2[i][0]][sparseArr2[i][1]] = sparseArr2[i][2];
        }
        System.out.println("稀疏数组转换成的二维数组");
        for (int[] rows : chessArray2) {
            for (int item : rows) {
                System.out.printf("%d\t",item);
            }
            System.out.println();
        }
        System.out.println();
    }
}
