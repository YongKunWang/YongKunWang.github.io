1. Terminal 实现文件复制操作

类似于Linux 下的 cp sources destination

package com.homewyk.thread.practice1;

import java.io.*;

/**
 * Q： 问题描述：
 *         类似于： Linux系统下的 cp source destination
 *         实现文件复制
 * @author asdw1
 *
 */
public class Demo1 {

    public static void main(String[] args) {
        /**
           * 判断输入参数
         */
        if (args.length != 2) {
            //System.out.println(args.length);
            System.out.println("Usage: xxx.exe source destination");
            return ;
        }
        try {
            cp(args[0],args[1]);
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
    /**
     * D：实现复制操作
     * @param source
     * @param destination
     * @throws IOException
     */
    public static void cp(String source, String destination) throws IOException {
        /**
         * D:判断参数
         *     1. 是否存在
         *  2. 是否是文件
         */
        File sourceFile = new File(source);
        if(!sourceFile.exists()) {
            System.out.println("File Not Find!!\n Please Check FileName");
            return ;
        }
        if(!sourceFile.isFile()) {
            System.out.println("FileName is not File...\n");
            return;
        }
        File destinationFile = new File(destination);
        if (!destinationFile.exists()) {
            destinationFile.createNewFile();
        }

        BufferedReader br = new BufferedReader(new FileReader(sourceFile));
        BufferedWriter bw = new BufferedWriter(new FileWriter(destinationFile));
        String str = null;
        while((str = br.readLine()) != null) {
            bw.write(str);
            bw.newLine();
            bw.flush();
        }
        if (bw != null) {
            bw.close();
        }
        if (br != null) {
            br.close();
        }
        System.out.println("-----------Copy finish--------");
    }
}

2. Terminal 实现文件夹的复制操作

类似于Linux下的 cp -r sourceFile destinatonFile

其实可以和上面的功能合二为一的

package com.homewyk.thread.practice2;

import java.io.*;

public class Demo {
    public static void main(String[] args) {
        if(args.length != 3) {
            System.out.println("Usage: java xxx -r sourceDirecttory destinationDirecttory");
            return ;
        }
        cpDir(new File(args[1]),new File(args[2]));
    }
    /**
     * D: 参数1： 源文件 其中包含文件夹&问价
     *         处理思路：
     *             1 当碰到文件夹时，destination也需要创建文件夹，还要进行迭代操作
     *             2.当碰到文件时，直接生成 
     *             3. 函数的参数最好为File类型，比较好弄！！进行第二步的时候直接传入就好了！
     * @param sourceFile
     * @param destinationFile
     */
    public static void cpDir(File sourceFile, File destinationFile) {

        if (sourceFile.isDirectory()) {
            if (!destinationFile.exists()) {
                destinationFile.mkdir();
            }
            if (sourceFile.isDirectory()) {
                String[] list = sourceFile.list();
                for (String string : list) {
                    File serFile = new File(sourceFile, string);
                    File destFile = new File(destinationFile, string);
                    cpDir(serFile, destFile);
                }
            }
        }else {
            try {
                cpFile(sourceFile, destinationFile);
            } catch (IOException e) {
                e.printStackTrace();
            }
        }
    }
    public static void cpFile(File source, File destination) throws IOException {

        BufferedReader br = new BufferedReader(new FileReader(source));
        BufferedWriter bw = new BufferedWriter(new FileWriter(destination));
        String str = null;
        while((str = br.readLine()) != null) {
            bw.write(str);
            bw.newLine();
            bw.flush();
        }
        if (bw != null) {
            bw.close();
        }
        if (br != null) {
            br.close();
        }
        //System.out.println("-----------Copy finish--------");
    }
}

3. 查找文件内容

在第二问的基础上，添加文件内容的查找

package com.homewyk.thread.practice3;

import java.io.*;
/***
 * D：当文件内部包含指定的字符串时，才复制相应的文件
 * @author asdw1
 *
 */
public class Demo {
    public static void main(String[] args) {
        if (args.length != 5) {
            System.out.println("Usage: java xxx find -r soucres destination findString");
            return;
        }
        File sourceFile = new File(args[2]);
        File destinationFile = new File(args[3]);
        try {
            cpDir(sourceFile, destinationFile, args[4]);
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
    /**
     * D： 遍历整个文件夹
     * @param sourceFile
     * @param destinationFile
     * @param str
     * @throws IOException
     */
    public static void cpDir(File sourceFile,File destinationFile, String str) throws IOException {
        if (sourceFile.isDirectory()) {
            if (!destinationFile.exists()) {
                destinationFile.mkdir();
            }
            String[] list = sourceFile.list();
            for (String string : list) {
                File srcFile = new File(sourceFile, string);
                File destFile = new File(destinationFile, string);
                cpDir(srcFile, destFile,str);
            }
        }else {
            cpFile(sourceFile, destinationFile, find(sourceFile, str));
        }
    }
    /**
     * D：判断文件中是否有指定的字符串，并且返回一个标志
     * @param sourceFile
     * @param str
     * @return
     * @throws IOException
     */
    public static boolean find(File sourceFile, String str) throws IOException {
        BufferedReader br = new BufferedReader(new FileReader(sourceFile));
        String string = null;
        while((string = br.readLine()) != null) {
            if (string.contains(str)) {
                return true;
            }
        }
        if (br != null) {
            br.close();
        }
        return false;
    }
    /**
     * D： 进行文件复制操作
     * @param sourceFile
     * @param destFile
     * @param flag
     * @throws IOException
     */
    public static void cpFile(File sourceFile, File destFile, boolean flag) throws IOException {
        if (flag) {
            BufferedReader br = new BufferedReader(new FileReader(sourceFile));
            BufferedWriter bw = new BufferedWriter(new FileWriter(destFile));
            String string = null;
            while((string = br.readLine()) != null) {
                bw.write(string);
                bw.newLine();
                bw.flush();
            }
            if (bw != null) {
                bw.close();
            }
            if (br != null) {
                br.close();
            }
        }
    }
}

5. 线程同步查找文件

遍历所有文件，当遍历到文件是.java的时候，创建一个线程去查找这个文件的内容，不必等待这个线程结束，继续遍历下一个文件 

package com.homewyk.thread.practice4;

import java.io.*;

/**
 * D: 实现线程同步查找特定文件类型中的内容，并输出绝对路径
 * @author asdw1
 *
 */

public class Demo1 {
    public static void main(String[] args) {
        if (args.length != 4) {
            System.out.println("Usage: java xxx find -r soucres findString");
            return;
        }
        File sourceFile = new File(args[2]);
        listDir(sourceFile, args[3]);
    }

    public static void listDir(File srcFile,String str) {
        if (srcFile.isDirectory()) {

            String[] list = srcFile.list(new MySuffix(".java"));
            for (String string : list) {
                File sourceFile = new File(srcFile,string);
                listDir(sourceFile, str);
            }
        }else {
            new Thread(new MyThread(srcFile, str)).start();
        }
    }
}

class MyThread implements Runnable {

    private File srcFile = null;
    private String str = null;

    public MyThread(File srcFile,String str) {
        this.srcFile = srcFile;
        this.str = str;
    }
    @Override
    public void run() {
        System.out.println("Thread-name: " + Thread.currentThread().getName());
        BufferedReader br = null;
        try {
            br = new BufferedReader(new FileReader(srcFile));
        } catch (FileNotFoundException e) {
            e.printStackTrace();
        }
        String string = null;
        try {
            while((string = br.readLine()) != null) {
                if (string.contains(str)) {
                    System.out.println(srcFile.getPath());
                }
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
        if (br != null) {
            try {
                br.close();
            } catch (IOException e) {
                e.printStackTrace();
            }
        }
    }

}

/**
 * D:实现文件类型过滤 
 *       文件夹直接通过
 * @author asdw1
 *
 */
class MySuffix implements FilenameFilter{
    private String suffix = null;

    public MySuffix(String suffix) {
        this.suffix = suffix;
    }

    @Override
    public boolean accept(File dir, String name) {
        File file = new File(dir,name);
        if(file.isFile()) {
            return name.endsWith(suffix);
        }else if (file.isDirectory()) {
            return true;
        }
        return false;
    }

}

6. 顺序线程的实现

英雄有可以放一个技能叫做: 波动拳-a du gen。

每隔一秒钟，可以发一次，但是只能连续发3次。

发完3次之后，需要充能5秒钟，充满，再继续发。

新建对象，使用此对象做为同步锁

使用匿名类，可以访问次外部变量。

使用了2个线程和一个标志位，标志位用来切换线程

也就是说，这两个线程的主要逻辑有了顺序！！

package com.homewyk.thread.practice5;

/***
 * 英雄有可以放一个技能叫做: 波动拳-a du gen。
 *    每隔一秒钟，可以发一次，但是只能连续发3次。
 *    
 *    发完3次之后，需要充能5秒钟，充满，再继续发。
 * 这里使用匿名类最好的！
 * @author asdw1
 *
 */
public class Demo {
    public static void main(String[] args) {
        Hero hero = new Hero("ddd", 0);
        (new Thread() {
            public void run() {
                while(true) {
                    synchronized (hero) {
                        System.out.println("Name:" + Thread.currentThread().getName());
                        if (hero.isFlag()) {
                            for (int i = 1; i <= hero.MAXSIZE; i++) {
                                System.out.println("释放技能：" + i + "次");
                                try {
                                    Thread.sleep(500);
                                } catch (InterruptedException e) {
                                    e.printStackTrace();
                                }
                            }
                            //技能释放完毕
                            hero.setFlag(false);
                            hero.notify();
                        }else { //等待冲能结束
                            try {
                                hero.wait();
                            } catch (InterruptedException e) {
                                e.printStackTrace();
                            }
                        }
                    }
                }

            }
        }).start();

        (new Thread() {
            public void run() {
                while(true) {
                    synchronized (hero) {
                        System.out.println("Name:" + Thread.currentThread().getName());
                        if (!hero.isFlag()) {
                            for (int i = 1; i <= 5; i++) {
                                System.out.println("技能冲能：" + i + "次");
                                try {
                                    Thread.sleep(500);
                                } catch (InterruptedException e) {
                                    e.printStackTrace();
                                }
                            }
                            //技能冲能完毕
                            hero.setFlag(true);
                            hero.notify();
                        }else { //等待技能释放结束
                            try {
                                hero.wait();
                            } catch (InterruptedException e) {
                                e.printStackTrace();
                            }
                        }
                    }
                }

            }
        }).start();

    }
}

class Hero {
    private String skill = null;
    private int counter = 0;
    final int MAXSIZE = 3;
    private boolean flag = true;

    public Hero(String skill, int counter) {
        this.skill = skill;
        this.counter = counter;
    }

    public String getSkill() {
        return skill;
    }

    public void setSkill(String skill) {
        this.skill = skill;
    }

    public int getCounter() {
        return counter;
    }

    public void setCounter(int counter) {
        this.counter = counter;
    }

    public boolean isFlag() {
        return flag;
    }

    public void setFlag(boolean flag) {
        this.flag = flag;
    }

    @Override
    public String toString() {
        return "Hero [skill=" + skill + ", counter=" + counter + ", MAXSIZE=" + MAXSIZE + ", flag=" + flag + "]";
    }
}
