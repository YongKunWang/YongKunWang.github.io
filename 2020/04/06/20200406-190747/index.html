过滤器Fliter

简介

  • 运行在服务器端的程序，先于与之先关的servlet和jsp之前运行，当对客户端进行访问时，对请求进行过滤：

    • 符合条件的，放行
    • 不符合条件的，不放行
    • 并且可以对目标资源访问前后进行逻辑处理

刚开始不明白没什么，当快速入门以后，在针对不会的进行处理

快速入门

步骤

  1. 编写一个过滤器的类实现Filter接口
    • 规范都是通过接口实现的
  2. 实现接口中尚未实现的方法(着重实现doFilter方法)
  3. 在web.xml中进行配置(主要是配置要对哪些目标资源进行过滤)

代码编写

  1. 先模仿一个目标资源：

    package com.homwyk.servlet;

import java.io.IOException;
import javax.servlet.ServletException;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

public class Servlet1 extends HttpServlet {

 protected void doGet(HttpServletRequest request, HttpServletResponse response)
         throws ServletException, IOException {

     System.out.println("Servlet1 running....");
     response.getWriter().write("Servlet1 running....");

 }

 protected void doPost(HttpServletRequest request, HttpServletResponse response)
         throws ServletException, IOException {
     doGet(request, response);
 }
}

  2. 编写过滤器：

    //和servletContext的生命周期一致
public class MyFilter implements Filter {
 @Override
 public void destroy() {
     System.out.println("destroy....");
 }

 @Override
 public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain)
         throws IOException, ServletException {
     System.out.println("MyFilter running ....");
     //程序运行到这，就相当于一块封闭的门，访问不到任何资源

     //这一步就变成了筛子
     chain.doFilter(request, response);

 }
 @Override
 public void init(FilterConfig filterConfig) throws ServletException {

     System.out.println("init....");
 }
}

  3. 在web.xml中进行配置

    <filter>
     <filter-name>MyFilter</filter-name>
     <filter-class>com.homwyk.filter.MyFilter</filter-class>
  </filter>
  <filter-mapping>
     <filter-name>MyFilter</filter-name>
     <url-pattern>/*</url-pattern>
  </filter-mapping>

Filter的访问流程

  • 多个Filter 可以称之为：Filter链
  • 有的目标资源不会被拦截

图片和上面相符合！！

Filter可以做的事情

  1. 在Filter中进行相关设置，目标资源可以获得
    • 公共代码的提取
    • 对request和response进行增强
      • 继承
      • 装饰者模式
      • 动态代理
  2. 可以使用1来全局编码
    • 在第一个Filter中设置编码格式，后面的都可以使用
  3. 权限管理：
    • 管理员能够访问全部资源
    • 用户能够获得部分资源

Filter的API介绍

Filter的生命周期

  • 在服务器启动的时候创建
  • 在服务器销毁的时候销毁
  • 每次有请求时会调用doFilter方法

Filter的xml的配置

 <filter>
    <filter-name>MyFilter</filter-name>
    <filter-class>com.homwyk.filter.MyFilter</filter-class>
  </filter>
  <filter-mapping>
    <filter-name>MyFilter</filter-name>
    <url-pattern>/*</url-pattern>
    <servlet-name>servlet1</servlet-name>
  </filter-mapping>

url-pattern的匹配方法：

  • 完全匹配 /servlet1
  • 文件名匹配 /aaa/*
  • 扩展名匹配 *.jsp
  • 注意文件名匹配和扩展名匹配不能混用/aa/*.jsp errrrrr

注意：url-pattern可以使用servlet-name替代，也可以混用

dispatcher：访问的方式

REQUEST:默认值，代表直接访问某个资源时执行filter

  • 转发不叫直接访问
  • 重定向才叫直接访问

FORWARD:转发时才执行filter，转发的时候才执行

INCLUDE：包含资源时执行filter e

ERROR：发生错误时 进行跳转是执行filter

案例一：实现自动登录

流程：

步骤一： 实现基本的登录操作

web主要逻辑：

public class Login extends HttpServlet {

    protected void doGet(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        HttpSession session = request.getSession();
        //1.获得网页端的参数
        String username = request.getParameter("username");
        String userpassword = request.getParameter("userpassword");
        
        Service service = new Service();
        User user = null;
        try {
            user = service.login(username,userpassword);
        } catch (SQLException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
        }
        if(user != null) {
            //登录成功，将对象存在Session中
            session.setAttribute("user", user);
            //重定向，会修改浏览器中显示的地址,重定向到index.jsp
            response.sendRedirect(request.getContextPath());
        }else {
            //登录失败，向request域中存放错误信息
            request.setAttribute("loginInfo", "用户名或者密码错误!");
            //转发，不改变浏览器中显示的地址
            request.getRequestDispatcher("/login.jsp").forward(request, response);
        }
    }
    protected void doPost(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        doGet(request, response);
    }
}

步骤二：实现自动登录

web:

public class Login extends HttpServlet {

    protected void doGet(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        
        HttpSession session = request.getSession();
        //1.获得网页端的参数
        String username = request.getParameter("username");
        String userpassword = request.getParameter("userpassword");
        
        //2. 是否自动登录
        String autoLogin = request.getParameter("autoLogin");
        
        
        Service service = new Service();
        User user = null;
        try {
            user = service.login(username,userpassword);
        } catch (SQLException e) {
            // TODO Auto-generated catch block
            e.printStackTrace();
        }
        if(user != null) {
            //自动登录的实现
            if(autoLogin != null) {
                Cookie cookie_username = new Cookie("username", user.getUsername());
                Cookie cookie_userpassword = new Cookie("userpassword", user.getUserpassword());
                //设置持久化时间
                cookie_username.setMaxAge(60*60);
                cookie_userpassword.setMaxAge(60*60);
                //设置cookie的携带路径
                cookie_username.setPath(request.getContextPath());
                cookie_userpassword.setPath(request.getContextPath());
                //发送cookie
                response.addCookie(cookie_username);
                response.addCookie(cookie_userpassword);
            }
            
            //登录成功，将对象存在Session中
            session.setAttribute("user", user);
            //重定向，会修改浏览器中显示的地址,重定向到index.jsp
            response.sendRedirect(request.getContextPath());
        }else {
            //登录失败，向request域中存放错误信息
            request.setAttribute("loginInfo", "用户名或者密码错误!");
            //转发，不改变浏览器中显示的地址
            request.getRequestDispatcher("/login.jsp").forward(request, response);
        }
    }

    protected void doPost(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
        doGet(request, response);
    }
}

filter:

public class AutoLogin implements Filter{

    @Override
    public void destroy() {     
    }

    @Override
    public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain)
            throws IOException, ServletException {
        //接收request请求中的cookie
        HttpServletRequest req = (HttpServletRequest) request;
        HttpServletResponse res = (HttpServletResponse) response;
        HttpSession session = req.getSession();
        
        String cookie_username = null;
        String cookie_userpassword = null;
        Cookie[] cookies = req.getCookies();
        for(Cookie cookie : cookies) {
            if("username".equals(cookie.getName())) {
                cookie_username = cookie.getValue();
            }
            if("userpassword".equals(cookie.getName())) {
                cookie_userpassword = cookie.getValue();
            }
        }
        if(cookie_username != null && cookie_userpassword != null) {
            Service service = new Service();
            User user = null;
            try {
                user = service.login(cookie_username,cookie_userpassword);
            } catch (SQLException e) {
                
                e.printStackTrace();
            }
            
            session.setAttribute("user", user);
        }
        //放行
        //是否自动登录都会经过这个filter
        // 非自动登录时，上面的全部都不执行，直接到Login中进行登录
        // 自动登录时，直接在这个filter中进行了登录与重定向的工作
        chain.doFilter(request, response);
    }

    @Override
    public void init(FilterConfig filterConfig) throws ServletException {       
    }

}

解决全局中文的问题：

使用装饰者模式

package com.homewyk.filter;

import java.io.IOException;
import java.io.UnsupportedEncodingException;

import javax.servlet.Filter;
import javax.servlet.FilterChain;
import javax.servlet.FilterConfig;
import javax.servlet.ServletException;
import javax.servlet.ServletRequest;
import javax.servlet.ServletResponse;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletRequestWrapper;

public class Encoding implements Filter{

    @Override
    public void destroy() {     
    }

    @Override
    public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain)
            throws IOException, ServletException {  
        //在传递request之前对request的getParameter方法进行增强
        /*
         * 装饰者模式(包装)
         * 
         * 1、增强类与被增强的类要实现统一接口
         * 2、在增强类中传入被增强的类
         * 3、需要增强的方法重写 不需要增强的方法调用被增强对象的
         * 
         */
        HttpServletRequest req = (HttpServletRequest)request;
        EnchanceEncoding enreq  = new EnchanceEncoding(req);
        chain.doFilter(enreq, response);
    }
    @Override
    public void init(FilterConfig filterConfig) throws ServletException {       
    }

}
class EnchanceEncoding extends HttpServletRequestWrapper{
    private HttpServletRequest req;
    
    public EnchanceEncoding(HttpServletRequest req) {
        super(req);
        this.req = req;
    }
    @Override
    public String getParameter(String name) {
        String paramter = req.getParameter(name);
        try {
            paramter =  new String(paramter.getBytes("iso8859-1"),"utf-8");
        } catch (UnsupportedEncodingException e) {
            e.printStackTrace();
        }
        return paramter;
    }
}
