  out = torch.zeros(label.size(0),depth)

  idx = torch.LongTensor(label).view(-1,1)

## 使用了3个线性层，2个relu激活函数
class Net(nn.Module):
  def __init__(self):
    super(Net,self).__init__()
    self.fc1 = nn.Linear(28*28,256)
    self.fc2 = nn.Linear(256,64)
    self.fc3 = nn.Linear(64,10)
  def forward(self, x):
    x = F.relu(self.fc1(x))
    x = F.relu(self.fc2(x))
    x = self.fc3(x)
    return x

net = Net()

def train(model, train_loader, optimizer, train_loss, epoch):
  # rain(mode=True)：将module设置为 training mode，只影响dropout和batchNorm
  model.train()
  for batch_idx, (data, target) in enumerate(train_loader):

    # 打平
    data = data.view(-1, 28*28)

    # 得到网络输出
    out = model(data)
    # one_hot编码
    y_one_hot = one_hot(target)

    # 使用MSE计算loss
    loss = F.mse_loss(out,y_one_hot)
    # grad 置0
    optimizer.zero_grad()
    # 反向梯度
    loss.backward()

    ## w = w` - lr *grad
    optimizer.step()

    ## item将tensor变为数值
    train_loss.append(loss.item())

    if batch_idx % 10 == 0:
      print(epoch, batch_idx, loss.item())

  plot_curve(train_loss)

optimizer = optim.SGD(net.parameters(),lr=0.01, momentum=0.9)

def plot_image(img, label, name):

    fig = plt.figure()
    for i in range(6):
        plt.subplot(2, 3, i + 1)
        plt.tight_layout()

        plt.title("{}: {}".format(name, label[i].item()))
        plt.xticks([])
        plt.yticks([])
    plt.show()

def one_hot(label, depth=10):
    out = torch.zeros(label.size(0), depth)
    idx = torch.LongTensor(label).view(-1, 1)
    out.scatter_(dim=1, index=idx, value=1)
    return out

def plot_curve(data):

class Net(nn.Module):
    def __init__(self):
        super(Net, self).__init__()

        # H1 = w1x+b1
        self.fc1 = nn.Linear(28 * 28, 256)
        # H2 = w2H1 + b2
        self.fc2 = nn.Linear(256, 64)
        # H3 = w3H2 + b3
        self.fc3 = nn.Linear(64, 10)
    def forward(self, x):
        x = F.relu(self.fc1(x))
        x = F.relu(self.fc2(x))
        x = self.fc3(x)

        return x

net = Net()

optimizer = optim.SGD(net.parameters(), lr=0.01, momentum=0.9)
train_loss = []
for epoch in range(3):
    for batch_idx , (x, y) in enumerate(train_loader):

         # input [b,1,28,28] y: 512
         # [b,1,28,28] => [b,784]
         # flatten
        x = x.view(x.size(0), 28*28)

        out = net(x)
        # 真实值的onehat编码
        y_onehat = one_hot(y)

        # 计算loss
        ## 使用MSE 平方差
        loss = F.mse_loss(out, y_onehat)
        # 梯度置零 必须
        optimizer.zero_grad()

        loss.backward()
        # w` = w - lr * grad
        optimizer.step()

        train_loss.append(loss.item())

        if batch_idx % 10 == 0:
            print(epoch, batch_idx, loss.item())

plot_curve(train_loss)

# 此时得到了一个比较好的[w1 b1 w2 b2 w3 b3]


total_currect = 0

for x, y in test_loader:

    # [512 1]

    # 正确的数量进行相加转换为float类型然后将张量的数值取出来
    currect = pred.eq(y).sum().float().item()
